Office.onReady(function (info) { if (info.host === Office.HostType.Word) { // Asegúrate de que los parámetros sean válidos Office.context.ui.addHandlerAsync( Office.EventType.DialogMessageReceived, function (args) { var message = args.message; console.log("Mensaje recibido del diálogo:", message); // Procesar el mensaje y actualizar el contenido dinámico if (message.title && message.proposals) { document.getElementById("dialogTitle").innerText = message.title; var proposalsContainer = document.getElementById("proposalsContainer"); proposalsContainer.innerHTML = ''; message.proposals.forEach(function (proposal) { var div = document.createElement('div'); div.innerHTML = `Propuesta ${proposal.id}: ${proposal.text}`; proposalsContainer.appendChild(div); }); } }, function (asyncResult) { // Este es el callback para manejar el resultado de la operación if (asyncResult.status === Office.AsyncResultStatus.Failed) { console.error("Error al agregar el manejador de eventos:", asyncResult.error.message); } } ); } });
